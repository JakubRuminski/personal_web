# from the socket module import all
from socket import *
from datetime import *

# -----------------------------------
# https://www.youtube.com/watch?v=3QiPPX-KeSc
""" Ideas regarding the use of threading for parallelism 
    where used from this video
     
    Also the method of sending the length of the message 
    first to the client and server were used   
"""
# -----------------------------------




sock = socket(AF_INET, SOCK_STREAM)

hostname = gethostname()
host_ip = gethostbyname(hostname)

server_address = (host_ip, 10000)

print("-------------\n"
      "Domain name: %s\n"
      "IP address %s\n"
      "-------------" % (hostname, host_ip))

print('*** Server is starting up on %s port %s ***' % server_address)
sock.bind(server_address)

sock.listen(1)

'''
 Places message fragment into logtxt
  
 @:param message The message to be logged
'''
def logMessage(message):
    file = open("logtxt", "a")
    message = datetime.now().strftime("%d/%m/%Y %H:%M ") + message
    file.writelines(message)
    file.close()
    return message


LENGTH = 16

while True:
    print('*** Waiting for a connection ***')
    connection, client_address = sock.accept()
    try:
        print('connection from', client_address)
        data_length = int(connection.recv(LENGTH).decode())
        # ^ first stream of bytes holds
        # the length of the message.
        message = ""
        data_received = 0
        while True:
            if data_received < data_length:
                data = connection.recv(LENGTH).decode()
                print('received "%s"' % data)
                message += data
                data_received += len(data)
            else:
                message += "\n"
                message = logMessage(message)
                message = f"{len(message):<{LENGTH}}" + message  # adds length fragment to let client know of length.
                connection.sendall(message.encode())
                print('no more data from', client_address)
                break
    finally:
        connection.close()
sock.close()
