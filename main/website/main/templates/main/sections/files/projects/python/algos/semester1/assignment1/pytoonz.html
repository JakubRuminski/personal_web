class Track:
    def __init__(self, __name, __artiste, __timesplayed):
        self.__name = __name
        self.__artiste = __artiste
        self.__timesplayed = __timesplayed
    def __str__(self):
        return "%s, %s, %d"% (self.get_name(), self.get_artiste(), self.__timesplayed)
    def get_name(self):
        return self.__name
    def get_artiste(self):
        return self.__artiste
    def play(self):
        timeplayed = int(self.__timesplayed)
        timeplayed += 1
        return "Playing: %s; %s; (%d)"% (self.__name, self.__artiste, timeplayed)

class DLLNode:
    def __init__(self, item, nextnode, prevnode):
        self.element = item
        self.next = nextnode
        self.previous = prevnode

class PyToonz:                                              # Tail --------- Head
    def __init__(self):
        self.tail = DLLNode(None, None, None)
        self.head = DLLNode(None, None, None)
        self.count = 0
        self.selection = None
    def __str__(self):
        current = self.tail.next
        string = "Playlist: \n"
        while current != self.head:
            if current == self.selection:
                string += "-->"
            string += str(current.element) + "\n"
            current = current.next
        return string
    def length(self):
        return self.count
    def add_track(self, track):
        track = track.split(", ")
        if len(track) == 3:
            if self.count == 0:
                node = DLLNode(track, self.head, self.tail)         # give node address of 'head' and then 'tail'.
                self.tail.next = node                               # gives tail.next address of node
                self.selection = node
            else:
                node = DLLNode(track, self.head, self.head.previous)
                #self.head.previous = node
                self.head.previous.next = node

            self.head.previous = node                               #give head the address of node.
            self.count = self.count + 1
            #print(node, node.previous, node.next)
        else:
            return None
    def get_current(self):
        if self.selection is not None:
            return "Current track: " + str(self.selection.element)
        else:
            return None
    def add_after(self, track):
        if self.count != 0:
            track = track.split(", ")
            node = DLLNode(track, self.selection.next, self.selection)
            self.selection.next.previous = node
            self.selection.next = node
            self.count = self.count + 1
        else:
            return self.add_track(track)
    def next_track(self):
        if self.count != 0:
            if self.selection.next == self.head:            # if last selection... wrap (check if working)  --[y]
                self.selection = self.tail.next
            else:
                self.selection = self.selection.next
        else:
            return None
    def prev_track(self):
        if self.count != 0:
            if self.selection.previous == self.tail:
                self.selection = self.selection             # if last... dont wrap (check if working)  --[y]
            else:
                self.selection = self.selection.previous
        else:
            return None
    def reset(self):
        if self.count > 0:
            self.selection = self.tail.next
    def play(self):
        if self.count > 0:
            t_details = self.selection.element
            return Track(t_details[0], t_details[1], t_details[2]).play()
        else:
            return "No tracks in list"
    def remove_current(self):
        if self.count > 0:
            self.selection.previous.next = self.selection.next
            self.selection.next.previous = self.selection.previous
            if self.count == 1:
                self.selection = None
            if self.selection.next == self.head:
                self.selection = self.selection.previous
            if self.selection.previous == self.tail:
                self.selection = self.selection.next
            self.count = self.count - 1
        else:
            return None
