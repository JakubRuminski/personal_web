# from the socket module import all
from socket import *
from datetime import *

sock = socket(AF_INET, SOCK_STREAM)

hostname = gethostname()
host_ip = gethostbyname(hostname)

print("-------------\n"
      "Domain name: %s\n"
      "IP address %s\n"
      "-------------" % (hostname, host_ip))

server_address = (host_ip, 10000)

print('connecting to server at %s port %s' % server_address)
sock.connect(server_address)

LENGTH = 16

try:
    message = input("Write message: ")
    print('sending "%s"' % message)

    message = f"{len(message):<{LENGTH}}" + message  # adds length fragment to let server know of length.
    sock.sendall(message.encode())

    data_length = int(sock.recv(LENGTH).decode())  # first message assumes server sent length.

    data_received = 0
    message = "received: "
    while True:
        if data_received < data_length:
            data = sock.recv(LENGTH).decode()
            message += data
            data_received += len(data)
        else:
            break
    print(message)


finally:
    print('closing socket')
    sock.close()
