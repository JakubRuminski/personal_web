# from the socket module import all
from socket import *

sock = socket(AF_INET, SOCK_STREAM)

ServerIP = input("Host name: ")
PortNo = int(input("Server port number: "))
FileName = input("File name: ")

##############################
# ServerIP = "192.168.178.25"
# PortNo = 6789
# FileName = "HelloWorld.html"
##############################

server_address = (ServerIP, PortNo)
print('connecting to server at %s port %s' % server_address)
sock.connect(server_address)

LENGTH = 16
GET_REQUEST = "GET /%s HTTP/11 Host: %s:%s"

try:
    message = GET_REQUEST % (FileName, ServerIP, PortNo)
    print('sending "%s"' % message)
    sock.sendall(message.encode())
    OK_message = sock.recv(16).decode()
    data_length = int(sock.recv(LENGTH).decode())  # first message assumes server sent length.

    data_received = 0
    message = "received: \n"
    while data_received < data_length:
        data = sock.recv(LENGTH).decode()
        message += data
        data_received += len(data)
    print(message)


finally:
    print('closing socket')
    sock.close()
