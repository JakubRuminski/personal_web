from tkinter import *
import .font as font
import random
import time

root = Tk()

canvas_x = 400
canvas_y = 400

main_bg = "lightgrey"

root.minsize(canvas_x+80, canvas_y+160)
root.configure(bg=main_bg)

class Time_Score:
    def __init__(self):
        self._time = 0
        self._score = 0
    def checktime(self):
        if time.time() - self._time < 2:
            self._score += 1
            label["text"] = "Score: " + str(self._score)
        self._time = time.time()
    def inc_score(self):
        self._score += 5
        if self._score == 5000:
            enemy_list.extend(enemy_list_gen())
        label["text"] = "Score: " + str(self._score)
class createCanvas:
    def __init__(self, r, size_x, size_y):
        self._size_x = size_x
        self._size_y = size_y
        self._root = r
        self.canvas = Canvas(self._root, width=self._size_x, height=self._size_y)
class Shape:
    def __init__(self):
        self.color = random.choice(["blue", "green", "red", "yellow", "orange", "purple"])
        self.size_x = random.randint(50, 120)
        self.size_y = random.randint(50, 120)
        self.x_pos = random.randint(1, canvas_x-self.size_x)
        self.y_pos = random.randint(1, canvas_y-self.size_y)
        self.object = None
class Player:
    def __init__(self):
        self.speed = 10
        self.y_pos = canvas_y//2
        self.height = 20
        self.width = 20
    def getDrawPlayer(self):
        return canvas_obj.create_rectangle(5, player.y_pos, 5 + player.width, player.y_pos + player.height, fill="green")
    def lost(self):
        self.color = "red"
        # Add--> stop movement once dead
class Enemy:
    def __init__(self):
        self.color = random.choice(["red", "black"])
        self.size = random.randint(15, 25)
        self.x_pos = canvas_x + random.randint(50, 75)
        self.y_pos = random.randint(1, canvas_y-self.size)
        self.speed = random.randint(3,5)
    def draw(self):
        if menuButton2["state"] == DISABLED:
            if self.x_pos + self.size < 0:
                self.x_pos = canvas_x + random.randint(50, 75)
                self.y_pos = random.randint(1, canvas_y - self.size)
                self.color = random.choice(["red", "black"])
                self.speed = random.randint(3,6)
            self.x_pos -= self.speed
            canvas_obj.create_oval(self.x_pos, self.y_pos, self.x_pos + self.size, self.y_pos + self.size, fill=self.color)


def collision():            # Checks if player was hit
    result = False
    for enemy in enemy_list:
        if (5 + player.width) < enemy.x_pos or \
        (player.y_pos + player.height) < enemy.y_pos or \
        5 > (enemy.x_pos + enemy.size) or \
        player.y_pos > (enemy.y_pos + enemy.size):
            result = False
        else:
            return True
    return result
def restart(event):             # Restarts Game 2
    if event.char == "r":
        for enemy in enemy_list:
            enemy.x_pos = canvas_x + random.randint(50, 75)
            enemy.y_pos = random.randint(0, canvas_y-enemy.size)
        stats._score = 0
        root.unbind("")
        game_rules_label["text"] = "Press 'E' to move up\nPress 'C' to move down"
        game2_canvas()

def game1_canvas():
    menuButton1["state"] = DISABLED
    root.unbind("")
    root.unbind("")
    button.pack_forget()
    canvas_obj.pack()
    label["text"] = "Score: 0"
    make(event=None)
def game2_canvas():
    clearcanvas()
    root.bind("", Check_Key)
    root.bind("", Check_Key)
    button.pack_forget()
    canvas_obj.pack()
def clearcanvas():
    canvas_obj.delete("all")
    player.getDrawPlayer()
    enemies()
    if collision():
        game_rules_label["text"] = "Game Over! \n Press 'R' to Restart"
        root.unbind("")
        root.unbind("")
        root.bind("", restart)
    elif menuButton2["state"] == DISABLED:
        stats.inc_score()
        root.after(33, clearcanvas)

def enemies():
    for enemy in enemy_list:
        enemy.draw()

def position(pos):
    if player.y_pos >= 0 and (player.y_pos + player.height <= canvas_y):
        player.y_pos += pos
    if player.y_pos < 0:
        player.y_pos = 0
    elif (player.y_pos + player.height) > canvas_y:
        player.y_pos = canvas_y - player.height

def Check_Key(event):
    if event.char == "e":
        position(-player.speed)
    elif event.char == "c":
        position(player.speed)


def make(event):
    canvas_obj.delete("all")
    shape.color = random.choice(["blue", "green", "red", "yellow", "orange", "purple"])
    shape.size_x = random.randint(50, 120)
    shape.size_y = random.randint(50, 120)
    shape.x_pos = random.randint(0, canvas_x-shape.size_x)
    shape.y_pos = random.randint(0, canvas_y-shape.size_y)

    stats.checktime()
    random.choice([make_oval, make_rectangle])()
def make_oval():
    shape.object = canvas_obj.create_oval(shape.x_pos,
                                    shape.y_pos,
                                    shape.x_pos+shape.size_x,
                                    shape.y_pos+shape.size_y,
                                    fill=shape.color)
    canvas_obj.tag_bind(shape.object, "", make)
def make_rectangle():
    shape.object = canvas_obj.create_rectangle(shape.x_pos,
                                          shape.y_pos,
                                          shape.x_pos + shape.size_x,
                                          shape.y_pos + shape.size_y,
                                          fill=shape.color)
    canvas_obj.tag_bind(shape.object, "", make)



def Change_to_Game_1():
    stats._score = 0
    canvas_obj.pack_forget()
    button.pack()                           # RePacks 'Main' button
    button["command"] = game1_canvas         # Change 'Main' button's function
    label["text"] = "Score: " + str(stats._score)
    game_rules_label.pack_forget()           # removes irrelevant Game2 instructions
    menuButton1["state"] = DISABLED         # Prevent multiple 'Game1' button presses
    menuButton2["state"] = NORMAL           # ReEnable 'Game2' button
def Change_to_Game_2():
    stats._score = 0
    button.pack()
    canvas_obj.pack_forget()
    game_rules_label.pack()
    button["command"] = game2_canvas
    label["text"] = "Score: " + str(stats._score)
    menuButton2["state"] = DISABLED
    menuButton1["state"] = NORMAL




main_font_size = font.Font(size=20)

frame = Frame(root)
menuButton1 = Button(frame, text="-Game1-", command=Change_to_Game_1, padx=30, font=main_font_size)
menuButton1.grid(row=1, column=1)
menuButton2 = Button(frame, text="-Game2-", command=Change_to_Game_2, padx=30, font=main_font_size)
menuButton2.grid(row=1, column=3)
frame.pack(side="top")


label = Label(root, text="To start press the button below!", bg=main_bg, font=main_font_size, pady=10)
label.pack()
button = Button(root, text="Click Me To Start!", command=game1_canvas, bg=main_bg, font=main_font_size)
button.pack()
game_rules_label = Label(root, text="Press 'E' to move up\nPress 'C' to move down", width=20, height=3, bg=main_bg)

canvas_obj = (createCanvas(root, canvas_x, canvas_y)).canvas
canvas_obj.configure(cursor="plus")

shape = Shape()
stats = Time_Score()

player = Player()

def enemy_list_gen():
    enemy_list = []
    i = 0
    while i < 10:
        enemy = Enemy()
        enemy_list.append(enemy)
        i += 1
    return enemy_list

enemy_list = enemy_list_gen()

root.mainloop()
