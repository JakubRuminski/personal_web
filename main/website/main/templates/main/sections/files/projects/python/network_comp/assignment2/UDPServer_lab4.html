from datetime import datetime
from socket import *

serverSocket = socket(AF_INET, SOCK_DGRAM)

serverPort = 6789
hostname = gethostbyname(gethostname())

server_address = (hostname, serverPort)
serverSocket.bind(server_address)

def logMessage(message):
    file = open("logtxt", "a")
    message = datetime.now().strftime("%d/%m/%Y %H:%M ") + message
    file.writelines(message)
    file.close()
    return message

while True:
    print("waiting on message from client")
    message, client_address = serverSocket.recvfrom(1024)
    print("Received message from ", client_address)
    message = message.decode()
    message = logMessage(message+"\n")
    print("message logged")
    try:
        message = message.upper()
    except ValueError:
        break
    print("message capitalized")
    message += "-END-"
    LENGTH = 16
    end = LENGTH
    start = 0
    while len(message) > end:
        serverSocket.sendto(message[start:end].encode(), client_address)
        start += LENGTH
        end += LENGTH
    serverSocket.sendto(message[start:end].encode(), client_address)
    print("Sending back logged info")

serverSocket.close()
